sync() example

var Product = Backbone.Model.extend({

   sync: function (method, model, options) {
       if (method === "read") {
           adapter.findById(this.id).done(function(data) {
               options.success(data);
           });
       } else if (method === "create") {
           adapter.addItem(this.attributes);
           this.id = this.get("id");
       }
   }

});
